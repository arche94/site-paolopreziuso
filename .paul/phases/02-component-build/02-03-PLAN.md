    **src/content.config.ts** (Astro v5 Content Layer API — file lives at `src/`, not `src/content/`):
    ```typescript
    import { defineCollection, z } from 'astro:content';
    import { glob } from 'astro/loaders';

    const work = defineCollection({
      loader: glob({ pattern: '**/*.yaml', base: './src/content/work' }),
      schema: z.object({
        order: z.number(),
        title: z.string(),
        period: z.string(),
        role: z.string(),
        company: z.string(),
        body: z.string(),
        stack: z.array(z.string()),
      }),
    });

    export const collections = { work };
    ```

    Fetch collection sorted by `order`, render section matching prototype Document view exactly.

    ```astro

    import { getCollection } from 'astro:content';
    const items = (await getCollection('work')).sort((a, b) => a.data.order - b.data.order);
    ---

    <section id="work">
      <div class="doc-section-label">
        <span class="num">§ 01</span> Selected work · 2018 — 2026
      </div>
      {items.map((item, i) => (
        <article class="doc-case">
          <div class="idx">N°&nbsp;{String(i + 1).padStart(2, '0')}</div>
          <div>
            <h3>{item.data.title}</h3>
            <div class="meta">{item.data.period} · {item.data.role} · {item.data.company}</div>
            <div class="body">{item.data.body}</div>
          </div>
          <div class="stack">
            {item.data.stack.map(tag => <span class="chip">{tag}</span>)}

        </article>
      ))}
    </section>
    ```
